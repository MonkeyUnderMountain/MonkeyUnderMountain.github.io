\section{Artin-Whaples approximations}

    \begin{theorem}[Artin-Whaples approximations]\label{theorem:artin_whaples_approximations}
        Let \(K\) be a field, and let \(v_1, v_2, \ldots, v_n\) be pairwise inequivalent nontrivial absolute values on \(K\). For any \(a_1, a_2, \ldots, a_n \in K\) and any \(\varepsilon > 0\), there exists an element \(x \in K\) such that
        \[
            |x - a_i|_{v_i} < \varepsilon
        \]
        for all \(1 \leq i \leq n\).

    \end{theorem}


\subsection{Geometric version}

    \begin{theorem}\label{thm:Artin_whaples_approximation_on_varieties}
        Let \(\kk\) be a field with algebraic closure \(\kkk\).
        Let \(X\) be a normal, projective, geometrically integral variety over \(\kk\).
        Let \(x_1, x_2, \ldots, x_n \in X(\kk)\) be closed points lying over pairwise distinct points of \(X\).
        Let \(v_1, v_2, \ldots, v_n \in M_\kk\) be pairwise inequivalent absolute values on \(\kk\).
        For every \(i = 1, 2, \ldots, n\), let \(U_i\) be an open neighborhood of \(x_i\) in \(X(\kkk)\) with respect to the topology induced by \(v_i\).
        Then there exists a rational point \(x \in X(\kk)\) such that \(x \in U_i\) for all \(1 \leq i \leq n\).

    \end{theorem}
